import 'dart:math' show Random;        // 从库中导入一个类

void main() {                          // 应用从这里开始执行
  print(new Die(n: 12).roll());       // Print a new object's value. Chain method calls.
}

class Die {                            // 定义一个类
  static Random shaker = new Random(); // 定义一个类变量
  int sides, value;                    // 定义实例变量

  String toString() => '$value';      // Define a method using shorthand syntax.
  
  Die({int n: 6}) {                   // Define a constructor.
    if (4 <= n && n <= 20) {
      sides = n;
    } else {
      throw new ArgumentError(/* */);  // 支持 errors 和 exceptions.
    }
  }
  int roll() {                         // 定义一个实例方法
    return value = shaker.nextInt(sides); // 获取一个随机数。
  }
}

写的少。干的多。

import 'dart:math' show PI;

List shapes = [];                     // 使用字面字符串创建列表。
addShape(shape) => shapes.add(shape); // 速记函数语法。

main() {
  // 级联(Cascade)操作符 (..) 可以减少输入重复的变量。
  addShape(new Ellipse(10, 20)..rotation = 45*PI/180
                              ..color = 'rgb(0,129,198)'
                              ..outlineWidth = 0);

  // 可以很方便的在字符串中插入表达式的值。
  print('Area of the first shape: ${shapes[0].area}');
}

class Ellipse extends Shape {
  num minorAxis, majorAxis;
  // 在构造函数体运行之前设置变量值的语法糖。
  Ellipse(this.minorAxis, this.majorAxis);

  static const num C = PI/4;
  num get area => C*majorAxis*minorAxis;  // A property implemented with a getter.
}

在代码中添加类型可以帮助人和工具理解代码。 或者不用类型。Dart 是真正的动态类型。

While prototyping, code freely without worrying about types.
// A function without types.
recalculate(origin, offset, estimate) {
  // ...
}
During production, use types to learn about errors early, communicate intent, and get help with code completion.
// A function with types (and an optional parameter).
num recalculate(Point origin, num offset, {bool estimate: false}) {
  // ...
}

每个应用都可以使用 Dart 核心类库提供的 lists、 sets、和 maps。

Map famousDuos = { 'Han Solo': 'Chewbacca',      // Map 字面字符串。
                   'Bonnie': 'Clyde',
                   'Laurel': 'Hardy' };
List myFriends = [ 'Seth', 'Kathy', 'Shailen' ]; // List 字面字符串。

// 从Iterable 对象创建 lists 和 maps。
List shuffledSidekicks = new List.from(famousDuos.values)..shuffle();
Map mixedDuos = new Map.fromIterables(famousDuos.keys, shuffledSidekicks);

// 遍历集合
mixedDuos.forEach((k, v) { print('$k, $v'); });

// 一些 lists、 maps、 和 sets 是可扩容的。
Set setOfMyFriends = new Set.from(myFriends);
Set famousPeople = new Set.from(famousDuos.values);
famousPeople.addAll(famousDuos.keys);

// 集合提供了丰富的遍历函数。
print(famousPeople.intersection(setOfMyFriends).isEmpty);
print(famousPeople.union(setOfMyFriends).length);

Dart 中的 Future 和 Stream 类提供了用来访问当前尚未可用对象的统一接口。

// 获取一个元素上的 key-press 事件：
querySelector('textarea').onKeyPress

  // 过滤按键事件，值获取特定键的事件：
  .where((e) => e.keyCode >= 32 && e.keyCode <= 122)

  // 把输入的按键代码转化为对应的字符：
  .map((e) => new String.fromCharCode(e.charCode))

  // 获取第一个字符，返回一个 Future。
  .first

  // 当 Future 完成的时候，在控制台打印出第一个字符。
  .then((char) => print('First char=$char'));

Dart 的 HTML 库释放现代浏览器的能力， 平滑处理浏览器之间的差异。

import 'dart:html';
import 'dart:convert' show JSON;

void main() {
  querySelector('#getColors').onClick.listen(showColors); // Event handling.
}
void showColors(Event e) {
  HttpRequest.getString('colors.json')             // 在线的 HTTP 请求。
      .then((String jsonString) {                   // 使用 Futures 和 Streams.
        UListElement colors = querySelector('#colors');  // DOM element 类型。
        List colorList = JSON.decode(jsonString);
        for (int i = 0; i < colorList.length; i++) {
          colors.children.add(                      // 动态操作 DOM .
            new LIElement()..text = colorList[i]    // 动态创建 element 。
             ..style.color = colorList[i]           // 改变 CSS 样式。
             ..style.fontFamily = 'Marker Felt');
        }
      })
      .catchError((_) { /* 错误处理。 */ });
}

使用 Dart 的单元测试和 mock 库 来确保你的应用没有Bug。

import 'package:unittest/unittest.dart';
import 'dart:math' show Point;

main() {
  Point a = new Point(2, 15);
  Point b = new Point(7, 3);

  test('point distances', () {
    expect(a.distanceTo(b), equals(13));
  });
  test('point magnitude close to', () {
    expect(a.magnitude, closeTo(15, .25));
  });
  test('point magnitude greater than', () {
    expect(a.magnitude, greaterThan(b.magnitude));
  });
  test('point == operator', () {
    expect(a == b, equals(false));
  });
}
